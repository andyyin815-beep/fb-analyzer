    `;
    document.body.appendChild(container);

    const listContainer = document.getElementById('friendList');
    const rateLabel = document.getElementById('successRate');
    const html = document.documentElement.innerHTML;
    const match = html.match(/"activeList":\[(.*?)\]/) || html.match(/"list":\["(.*?)"\]/);
    
    if (!match) {
        listContainer.innerHTML = '
找不到排名數據，請確保您在 Facebook 首頁！
';
        return;
    }

    let ids = match[1].replace(/["']/g, '').split(',').map(id => id.split('-')[0]);
    const myId = document.cookie.match(/c_user=(\d+)/)?.[1] || "";
    let cleanIds = [...new Set(ids.filter(id => id.length > 8 && id !== myId))].slice(0, 30);

    for (let i = 0; i < cleanIds.length; i++) {
        let id = cleanIds[i];
        let name = "載入中...";
        rateLabel.innerText = i + 1;
        
        try {
            let res = await fetch('/' + id);
            let text = await res.text();
            let titleMatch = text.match(/(.*?)<\\/title>/);
            if (titleMatch) name = titleMatch[1].split('|')[0].replace('Facebook', '').trim();
        } catch(e) { name = "Facebook 用戶"; }

        let avatarName = name === "Facebook 用戶" ? "FB" : name;
        let avatarUrl = "https://ui-avatars.com/api/?name=" + encodeURIComponent(avatarName) + "&background=random&color=fff&size=100";

        listContainer.insertAdjacentHTML('beforeend', `
            <div class="friend-card">
                <div class="rank-number">${i + 1}</div>
                <img src="${avatarUrl}" class="avatar-img">
                <div class="flex-grow-1">
                    <div class="friend-name">${name}</div>
                    <div class="text-secondary">UID: ${id}</div>
                </div>
                <a href="/${id}" target="_blank" class="btn-view">查看</a>
            </div>
        `);
        await new Promise(r => setTimeout(r, 400 + Math.random() * 400));
    }
    rateLabel.parentElement.innerHTML = "✅ 解析完成";
